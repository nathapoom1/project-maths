A0 [2 3 -2 7]    = B0 [-11]
A1 [5 0 4 0]     = B1 [37]
A2 [0 2 5 -4]    = B2 [31]
A3 [6 2 3 5]     = B3 [28]

for i in range(n):
    if A[i, i] == 0:
        for k in range(i+1, n):
            if A[k, i] != 0:
                A[[i, k]] = A[[k, i]]
                b[[i, k]] = b[[k, i]]
                break
        else:
            raise ValueError("Matrix is singular.")
